CREATE DATABASE Praticando;

USE Praticando;

-- Tabela Clientes
CREATE TABLE Clientes (
	IdCliente INT PRIMARY KEY, 
	Nome VARCHAR(50)
);

-- Tabela Pedidos
CREATE TABLE Pedidos (
	IdPedido INT PRIMARY KEY,
	IdCliente INT,
	Produto VARCHAR(50)
);
